#安装完git后， 需要设置，在命令行输入：
git config --global user.name "Your Name"
git config --global user.email "email@example.com"

#创建版本库
1.在合适的地方，创建文件夹
sudo mkdir ./git
cd ./git
pwd

2.通过命令git init 把这个目录变为Git可以管理的仓库
git init


#往仓库添加文件
1.在仓库里创建文件并写入内容
2.git add filename
3.git commit -m "备注说明"

$例子
git add file1.txt
git add file2.txt file3.txt
git commit -m "add 3 files."

#时光穿梭机 查看修改
git status
(这个命令可以掌握仓库当前的状态)
git diff filename
(该命令可以查看文件做了什么修改)
接下来可以提交了，两步add  commit
